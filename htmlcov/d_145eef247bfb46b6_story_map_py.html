175 # 스토리 확인 창이 나오는 중에는 스토리 선택은 작동하지 않게 처리 

176 global enter_story

179 for i in range(self.max_menu): 

180 if event.type == pygame.MOUSEBUTTONDOWN:

182 self.select_menu(i) 

183 break # 한 번에 여러 개의 메뉴가 눌리지 않도록 처리 

184 elif event.type == pygame.MOUSEMOTION: 

185 if self.rect[i].collidepoint(event.pos): 

186 # 해금되지 않은 스토리에는 반응하지 않되, '돌아가기'에는 반응 

187 if (i <= StoryMenu.story_progress) or i == self.max_menu - 1: 

188 # highlight 대상을 변경 

189 self.highlight = i 

190 # 키보드 선택 해제 

191 self.selected = -1 

192 elif event.type == pygame.KEYDOWN: 

193 if self.pressed == False: 

194 self.pressed = True 

195 # 엔터 키가 눌렸을 때 

196 if event.key == setting.options["enter"]: 

197 # 키보드로 선택한 것이 있다면 그 메뉴를 선택 

198 if self.selected != -1: 

199 self.select_menu(self.selected) 

200 elif ( 

201 event.key 

202 == setting.options["up" if self.axis == "y" else "left"] 

203 ): 

204 # 선택을 하나 위로 이동 

205 self.selected = self.selected - 1 if 0 < self.selected else 0 

206 # 돌아가기에서 미해금 스토리로 이동 시, (예: 돌아가기 -> 5번) 

207 # 해금된 가장 마지막 스토리로 이동 (예: 5번 -> 3번) 

208 if self.selected > StoryMenu.story_progress: 

209 self.selected = StoryMenu.story_progress 

210 self.highlight = self.selected 

211 elif ( 

212 event.key 

213 == setting.options["down" if self.axis == "y" else "right"] 

214 ): 

215 # 선택을 하나 아래로 이동 

216 self.selected = (

220 ) 

221 # 해금된 마지막 스토리에서 미해금으로 이동 시, (예: 3번 -> 4번)

225 self.highlight = self.selected

227 elif event.type == pygame.KEYUP:
